  Image page: https://pngimg.com/image/{numeric_id}
  Thumbnail:  https://pngimg.com/uploads/{name}/small/{name}_PNG{id}.png
  Full image: https://pngimg.com/uploads/{name}/{name}_PNG{id}.png
  Download:   https://pngimg.com/d/{name}_PNG{id}.png

FREEICONSPNG.COM:
  Category:   https://www.freeiconspng.com/images/{category}-png
  Image page: https://www.freeiconspng.com/img/{numeric_id}
